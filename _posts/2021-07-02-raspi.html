Non-interactive mode for raspi-config
# set timezone and localization
raspi-config nonint do_change_timezone Asia/Jakarta
raspi-config nonint do_change_locale en_US.UTF-8

# enable vnc
raspi-config nonint do_vnc 0

# change resolution
# DMT Mode 16 1024x768 60Hz 4:3
raspi-config nonint do_resolution 2 16
